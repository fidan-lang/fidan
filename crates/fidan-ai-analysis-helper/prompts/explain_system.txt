Fidan language reference (compact — all syntax uses braces, never `end`):

Comments:
  # single-line comment
  ## also single-line
  #/ multi-line  (nestable)  /#

Imports:
  use std.io
  use std.io.print
  use std.io.{print, readFile}
  use std.math as math
  export use std.io.print               # re-export

Built-ins:
  integer (int)
  float (decimal)
  boolean (bool)
  string (text)
  nothing
  flexible
  len(obj)
  type(obj)
  print(obj) # stdout
  eprint(obj)  # stderr
  panic(msg)   # raises error

Variables (= and set are interchangeable):
  var x = 42                            # inferred integer
  var b set 3.14                        # inferred float
  var s = "hello"                       # inferred string
  var flag = true                       # inferred boolean
  var n = nothing                       # null
  var f oftype integer                  # typed, no init (= nothing)
  var g oftype float = 2.718            # typed + init
  var x -> list -> integer              # arrow syntax = `oftype list oftype integer`

Type keywords: integer (int), float (decimal), boolean (bool), string (text), nothing, flexible
Composite: list oftype T, map oftype T, dict oftype (K, V), Shared oftype T, Pending oftype T, WeakShared oftype T

Constants:
  const MAX = 100

Actions (functions) — always brace-delimited:
  action greet with (certain name oftype string) returns string {
      return "Hello, {name}!"
  }
  action greetOptional with (optional username oftype string = "world") {
      print("Hello, {username ?? "stranger"}!")
  }
  # `certain` = non-nullable param; `optional` = nullable with optional default
  # `also` = `,` alias in param and arg lists
  action add with (certain x -> integer also certain y -> integer) returns integer {
      return x + y
  }
  # parallel action — returns Pending oftype T automatically
  parallel action fetchData with (certain url oftype string) returns string { ... }
  # extension action — callable as Dog method or free function
  action greet extends Dog with (optional guest oftype Dog) { ... }

  # Anonymous action (lambda) — inline, first-class value; use `action` NOT `function`:
  var double = action with (certain x oftype integer) returns integer { return x * 2 }
  var noop   = action { print("done") }
  list.forEach(action with (certain item oftype string) { print(item) })

Objects — brace-delimited, constructor uses `new with` (no `action` keyword), `this` not `self`:
  object Point {
      var x oftype integer
      var y oftype integer
      new with (certain x oftype integer, certain y oftype integer) {
          this.x = x
          this.y = y
      }
      action to_string returns string {
          return "({this.x}, {this.y})"
      }
  }
  # inheritance
  object Dog extends Animal {
      new with (certain name oftype string) {
          parent(species set "Dog", sound set "Woof")
          this.name = name
      }
  }

Enums — brace-delimited:
  enum Color { Red  Green  Blue }

Decorators:
  @precompile          # eager JIT warm-up
  @deprecated
  @gpu                 # GPU kernel (hard error if no GPU)

Control flow — always brace-delimited, no parens on condition:
  if score >= 90 {
      print("A")
  } otherwise when score >= 80 {   # `otherwise when` = else if; `else if` also valid
      print("B")
  } otherwise {                    # `otherwise` = else
      print("F")
  }

  for i in 1...10 { ... }          # inclusive end (1,2,...,10)
  for i in 1..10  { ... }          # exclusive end (1,2,...,9)
  for item in list { ... }         # iterate list
  while cond { ... }
  break  /  continue  /  stop      # `stop` = `break` alias

Check (switch) — brace-delimited, `_` or `default` for wildcard:
  check value {
      "Monday" => { print("Start") }
      "Friday" => { print("End")   }
      _        => { print("Other") }
  }
  # check as expression:
  var label = check code { 200 => "OK"  404 => "Not Found"  _ => "Other" }

Error handling:
  attempt {
      panic("oops")          # `throw` = alias
  } catch err {
      print("caught: {err}")
  } catch err oftype string {
      print("string error: {err}")
  } catch err -> CustomError {
      print("typed: {err}")
  } otherwise {
      print("no error — success path")
  } finally {
      print("always runs")
  }

Concurrency:
  concurrent { task A { ... } task B { ... } }   # cooperative, same thread
  parallel   { task A { ... } task B { ... } }   # OS threads via Rayon
  parallel for item in data { ... }               # parallel iteration
  var handle = spawn fetchData("url")             # Pending oftype string
  var result = await handle                       # unwrap to string

Shared state:
  var counter oftype Shared oftype integer = Shared(0)
  counter.get()  /  counter.set(val)  /  counter.weak()  # → WeakShared
  var w oftype WeakShared oftype integer = counter.weak()
  w.upgrade()    # → Shared oftype T

Calls — named args use `set` or `=`, `also` = `,`:
  add(3, 4)
  add(x set 3, y set 4)
  add(x = 3, y = 4)
  createUser("Alice" also age = 25)

Operators:
